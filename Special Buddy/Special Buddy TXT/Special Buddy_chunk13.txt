✅ 1. BuildDatabaseFromScript() Function
Here's a clean, VBA-friendly function that reads an SQL file and executes it statement by statement (based on semicolons):

Public Function BuildDatabaseFromScript(ByVal sqlFilePath As String) As Boolean
    On Error GoTo ErrHandler

    Dim fileNum As Integer
    Dim fileContent As String
    Dim sqlStatements() As String
    Dim sqlLine As Variant

    fileNum = FreeFile()
    Open sqlFilePath For Input As #fileNum
    fileContent = Input$(LOF(fileNum), fileNum)
    Close #fileNum

    ' Split by semicolon
    sqlStatements = Split(fileContent, ";")

    Dim executedCount As Long
    executedCount = 0

    For Each sqlLine In sqlStatements
        sqlLine = Trim(sqlLine)
        If Len(sqlLine) > 0 Then
            Debug.Print "Running SQL: " & Left(sqlLine, 80) & "..."
            CurrentDb.Execute sqlLine & ";", dbFailOnError
            executedCount = executedCount + 1
        End If
    Next sqlLine

    MsgBox "Executed " & executedCount & " SQL statement(s).", vbInformation
    BuildDatabaseFromScript = True
    Exit Function

ErrHandler:
    MsgBox "Error: " & Err.Description & vbCrLf & "Statement: " & sqlLine, vbCritical
    BuildDatabaseFromScript = False
End Function
🧠 Notes:

Assumes the script is clean and doesn’t contain multi-semi-colon statements on one line.

You can extend it to log errors or skipped lines.
